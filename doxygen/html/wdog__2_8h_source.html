144 /* Disable the watchdog by clearing WDE. That special sequence must be
145  * respected.
146  *
147  * FIXME: that also clears WDRF (required) and other reset flags, as well
148  * as WDIE but this should not be a problem. WDP and WDIF are left intact.
149  * May be should only clear WDRF
150  *
151  * FIXME: 0x27 is the mask for WDP bits in WDTCR
152  */
153 #define _hw_turn_wdog_off(c,n,i,a) \
154  do { \
155  uint8_t reg ; \
156  _hw_write_reg( hw_core0, mcusr, 0 ); \
157  __asm__ __volatile__(" in %[r], %[wdtcr]" "\n\t" \
158  " ori %[r], %[wdce]|%[wde]" "\n\t" \
159  " out %[wdtcr], %[r]" "\n\t" \
160  " andi %[r], %[wdp]" "\n\t" \
161  " out %[wdtcr], %[r]" "\n\t" \
162  : [r] "=&d" (reg) \
163  : [wdtcr] "I" (_hw_addr(_hw_reg(c,n,i,a, csr))-0x20), \
164  [wdce] "I" (1<<_hw_bp(_hw_reg(c,n,i,a, wdce))), \
165  [wde] "I" (1<<_hw_bp(_hw_reg(c,n,i,a, wde))), \
166  [wdp] "I" (0x27)); \
167  } while(0)
