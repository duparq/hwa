224 /* Solve the configuration of the counter with its compare and capture units
225  */
226 HW_INLINE void __hwa_solve__c16a ( hwa_t *hwa __attribute__((unused)), hwa_c16a_t *p )
227 {
228  p->solved.cs = 0xFF ;
229  p->solved.wgm = 0xFF ;
230  p->solved.coma = 0xFF ;
231  p->solved.comb = 0xFF ;
232  p->solved.acic = 0xFF ;
233  p->solved.ices = 0xFF ;
234  p->solved.icnc = 0xFF ;
235 
236  if ( p->config.clock == 0xFF )
237  return ;
238 
239  /* Clock setting
240  */
241  p->solved.cs = p->config.clock ;
242 
243  /* Default config for top
244  */
245  if ( p->config.top == 0xFF )
246  p->config.top = HW_A1(hw_c16a_top_max);
247 
248  /* Default config for overflow
249  */
250  uint8_t overflow = p->config.overflow ;
251  if ( overflow == 0xFF && p->config.top == HW_A1(hw_c16a_top_compare0) ) {
252  if ( p->config.countmode == HW_A1(hw_c16a_countmode_loop_up) )
253  overflow = HW_A1(hw_c16a_overflow_at_top);
254  else /* if ( p->config.countmode == HW_A1(hw_c16a_countmode_loop_up) ) */
255  overflow = HW_A1(hw_c16a_overflow_at_bottom);
256  }
257 
258  /* Compare update setting
259  */
260  uint8_t compare_update = 0xFF ;
261  if ( p->config.compare0.update != 0xFF && p->config.compare1.update != 0xFF
262  && p->config.compare0.update != p->config.compare1.update )
263  HWA_ERR("optionnal parameter 'update' of class _c16a counter must be " \
264  "'immediately'.");
265  compare_update = p->config.compare0.update ;
266 
267  /* Mode WGM Operation COUNTMODE TOP OCR OVF OCF ICF OCA OCB
268  * TOP
269  * UPD
270  *
271  * 0 0000 Normal LOOP_UP 0xFFFF IMM MAX CoM,SoM,ToM CoM,SoM,ToM
272  * 4 0100 CTC (Clear Timer on Match) LOOP_UP ocra IMM MAX TOP CoM,SoM,ToM
273  * 12 1100 CTC (Clear Timer on Match) LOOP_UP icr IMM MAX TOP CoM,SoM,ToM CoM,SoM,ToM
274  *
275  * 5 0101 Fast PWM, 8-bit LOOP_UP 0xFF TOP TOP SaBCoM,CaBSoM SaBCoM,CaBSoM
276  * 6 0110 Fast PWM, 9-bit LOOP_UP 0x1FF TOP TOP SaBCoM,CaBSoM SaBCoM,CaBSoM
277  * 7 0111 Fast PWM, 10-bit LOOP_UP 0x3FF TOP TOP SaBCoM,CaBSoM SaBCoM,CaBSoM
278  * 14 1110 Fast PWM LOOP_UP icr TOP TOP TOP ToM,SaBCoM,CaBSoM SaBCoM,CaBSoM
279  * 15 1111 Fast PWM LOOP_UP ocra TOP TOP TOP SaBCoM,CaBSoM
280  *
281  * 1 0001 PWM, Phase Correct, 8-bit LOOP_UPDOWN 0xFF TOP BOT CMUSMD,SMUCMD CMUSMD,SMUCMD
282  * 2 0010 PWM, Phase Correct, 9-bit LOOP_UPDOWN 0x1FF TOP BOT CMUSMD,SMUCMD CMUSMD,SMUCMD
283  * 3 0011 PWM, Phase Correct, 10-bit LOOP_UPDOWN 0x3FF TOP BOT CMUSMD,SMUCMD CMUSMD,SMUCMD
284  * 10 1010 PWM, Phase Correct LOOP_UPDOWN icr TOP BOT TOP ToM,CMUSMD,SMUCMD CMUSMD,SMUCMD
285  * 11 1011 PWM, Phase Correct LOOP_UPDOWN ocra TOP BOT TOP CMUSMD,SMUCMD
286  *
287  * 8 1000 PWM, Phase & Freq. Correct LOOP_UPDOWN icr BOT BOT TOP ToM,CMUSMD,SMUCMD CMUSMD,SMUCMD
288  * 9 1001 PWM, Phase & Freq. Correct LOOP_UPDOWN ocra BOT BOT TOP CMUSMD,SMUCMD
289  *
290  * 13 1101 *Reserved*
291  */
292 
293  /* Determine WGM
294  */
295  uint8_t wgm = 0xFF ;
296  if ( p->config.countmode == HW_A1(hw_c16a_countmode_loop_up)
297  && p->config.top == HW_A1(hw_c16a_top_compare0)
298  && overflow == HW_A1(hw_c16a_overflow_at_top) )
299  wgm = 15 ;
300  else
301  if ( p->config.countmode == HW_A1(hw_c16a_countmode_loop_up) ) {
302  if ( p->config.top == HW_A1(hw_c16a_top_fixed_0xFFFF) )
303  wgm = 0 ;
304  else if (p->config.top == HW_A1(hw_c16a_top_fixed_0xFF) )
305  wgm = 5 ;
306  else if (p->config.top == HW_A1(hw_c16a_top_fixed_0x1FF) )
307  wgm = 6 ;
308  else if (p->config.top == HW_A1(hw_c16a_top_fixed_0x3FF) )
309  wgm = 7 ;
310  else if (p->config.top == HW_A1(hw_c16a_top_compare0) ) {
311  if ( compare_update == HW_A1(hw_ocu_update_immediately)
312  || overflow == HW_A1(hw_c16a_overflow_at_top)
313  || p->config.compare1.output == HW_A1(hw_ocu_output_set_at_bottom_clear_on_match)
314  || p->config.compare1.output == HW_A1(hw_ocu_output_clear_at_bottom_set_on_match))
315  wgm = 15 ;
316  else
317  wgm = 4 ;
318  }
319  else /* if (p->top == HW_A1(hw_c16a_top_capture0) ) */ {
320  if (compare_update == HW_A1(hw_ocu_update_at_top)
321  || overflow == HW_A1(hw_c16a_overflow_at_top)
322  || p->config.compare0.output == HW_A1(hw_ocu_output_toggle_on_match)
323  || p->config.compare0.output == HW_A1(hw_ocu_output_set_at_bottom_clear_on_match)
324  || p->config.compare0.output == HW_A1(hw_ocu_output_clear_at_bottom_set_on_match)
325  || p->config.compare1.output == HW_A1(hw_ocu_output_set_at_bottom_clear_on_match)
326  || p->config.compare1.output == HW_A1(hw_ocu_output_clear_at_bottom_set_on_match))
327  wgm = 14 ;
328  else
329  wgm = 12 ;
330  }
331  }
332  else {
333  if (p->config.top == HW_A1(hw_c16a_top_fixed_0xFF) )
334  wgm = 1 ;
335  else if (p->config.top == HW_A1(hw_c16a_top_fixed_0x1FF) )
336  wgm = 2 ;
337  else if (p->config.top == HW_A1(hw_c16a_top_fixed_0x3FF) )
338  wgm = 3 ;
339  else if (p->config.top == HW_A1(hw_c16a_top_compare0) ) {
340  if (compare_update == HW_A1(hw_ocu_update_at_bottom)
341  || p->config.compare1.output == HW_A1(hw_ocu_output_clear_on_match_up_set_on_match_down)
342  || p->config.compare1.output == HW_A1(hw_ocu_output_set_on_match_up_clear_on_match_down))
343  wgm = 9 ;
344  else
345  wgm = 11 ;
346  }
347  else /* if (p->top == HW_A1(hw_c16a_top_capture0) ) */ {
348  if (compare_update == HW_A1(hw_ocu_update_at_bottom)
349  || p->config.compare0.output == HW_A1(hw_ocu_output_toggle_on_match)
350  || p->config.compare0.output == HW_A1(hw_ocu_output_clear_on_match_up_set_on_match_down)
351  || p->config.compare0.output == HW_A1(hw_ocu_output_set_on_match_up_clear_on_match_down)
352  || p->config.compare1.output == HW_A1(hw_ocu_output_clear_on_match_up_set_on_match_down)
353  || p->config.compare1.output == HW_A1(hw_ocu_output_set_on_match_up_clear_on_match_down))
354  wgm = 8 ;
355  else
356  wgm = 10 ;
357  }
358  }
359 
360  if (wgm != 0xFF)
361  p->solved.wgm = wgm ;
362  else {
363  HWA_ERR("WGM value could not be solved for _c16a class counter.");
364  return ;
365  }
366 
367  /* Solve the configuration of compare output A
368  */
369  if ( p->config.compare0.output != 0xFF ) {
370 
371  uint8_t mode = 0xFF ;
372 
373  if ( p->config.compare0.output == HW_A1(hw_ocu_output_disconnected) )
374  mode = 0 ;
375  else if ( p->config.compare0.output == HW_A1(hw_ocu_output_toggle_on_match) )
376  mode = 1 ;
377  else if ( p->config.compare0.output == HW_A1(hw_ocu_output_clear_on_match)
378  || p->config.compare0.output == HW_A1(hw_ocu_output_set_at_bottom_clear_on_match)
379  || p->config.compare0.output == HW_A1(hw_ocu_output_clear_on_match_up_set_on_match_down) )
380  mode = 2 ;
381  else
382  mode = 3 ;
383 
384  p->solved.coma = mode ;
385  }
386 
387  /* Solve the configuration of compare output B
388  */
389  if ( p->config.compare1.output != 0xFF ) {
390 
391  uint8_t mode = 0xFF ;
392 
393  if ( p->config.compare1.output == HW_A1(hw_ocu_output_disconnected) )
394  mode = 0 ;
395  else if ( p->config.compare1.output == HW_A1(hw_ocu_output_toggle_on_match) )
396  mode = 1 ;
397  else if ( p->config.compare1.output == HW_A1(hw_ocu_output_clear_on_match)
398  || p->config.compare1.output == HW_A1(hw_ocu_output_set_at_bottom_clear_on_match)
399  || p->config.compare1.output == HW_A1(hw_ocu_output_clear_on_match_up_set_on_match_down) )
400  mode = 2 ;
401  else
402  mode = 3 ;
403 
404  p->solved.comb = mode ;
405  }
406 
407  /* Solve the configuration of the capture input
408  */
409  if ( p->config.capture0.input != 0xFF ) {
410  p->solved.acic = p->config.capture0.input-1 ;
411  p->solved.ices = p->config.capture0.edge-1 ;
412  if ( p->config.capture0.filter != 0xFF )
413  p->solved.icnc = p->config.capture0.filter ;
414  }
415 
416 
417  /* Check the validity of the configuration
418  */
419  if ( p->config.countmode != 0xFF || p->config.compare0.output != 0xFF || p->config.compare1.output != 0xFF ) {
420 
421  if ( p->config.countmode == 0xFF ) {
422  HWA_ERR("configuration of counter is required.");
423  return ;
424  }
425 
426  /* Check compare output A
427  */
428  if ( p->config.compare0.output != 0xFF ) {
429  if ( wgm==0 || wgm==12 ) {
430  if ( p->config.compare0.output != HW_A1(hw_ocu_output_disconnected)
431  && p->config.compare0.output != HW_A1(hw_ocu_output_clear_on_match)
432  && p->config.compare0.output != HW_A1(hw_ocu_output_set_on_match)
433  && p->config.compare0.output != HW_A1(hw_ocu_output_toggle_on_match) )
434  HWA_ERR("compare output A of class _c16a counter mode must be "
435  "'disconnected', 'clear_on_match', 'set_on_match', or "
436  "'toggle_on_match'.");
437  }
438  else if ( wgm==5 || wgm==6 || wgm==7 ) {
439  if ( p->config.compare0.output != HW_A1(hw_ocu_output_disconnected)
440  && p->config.compare0.output != HW_A1(hw_ocu_output_set_at_bottom_clear_on_match)
441  && p->config.compare0.output != HW_A1(hw_ocu_output_clear_at_bottom_set_on_match) )
442  HWA_ERR("compare output A of class _c16a counter mode must be "
443  "'disconnected', 'set_at_bottom_clear_on_match', or "
444  "'clear_at_bottom_set_on_match'.");
445  }
446  else if ( wgm==14 ) {
447  if ( p->config.compare0.output != HW_A1(hw_ocu_output_disconnected)
448  && p->config.compare0.output != HW_A1(hw_ocu_output_toggle_on_match)
449  && p->config.compare0.output != HW_A1(hw_ocu_output_set_at_bottom_clear_on_match)
450  && p->config.compare0.output != HW_A1(hw_ocu_output_clear_at_bottom_set_on_match) )
451  HWA_ERR("compare output A of class _c16a counter mode must be "
452  "'disconnected', 'toggle_on_match', "
453  "'set_at_bottom_clear_on_match', or "
454  "'clear_at_bottom_set_on_match'.");
455  }
456  else if ( wgm==1 || wgm==2 || wgm==3 ) {
457  if ( p->config.compare0.output != HW_A1(hw_ocu_output_disconnected)
458  && p->config.compare0.output != HW_A1(hw_ocu_output_clear_on_match_up_set_on_match_down)
459  && p->config.compare0.output != HW_A1(hw_ocu_output_set_on_match_up_clear_on_match_down) )
460  HWA_ERR("compare output A of class _c16a counter mode must be "
461  "'disconnected', 'clear_on_match_up_set_on_match_down', "
462  "or 'set_on_match_up_clear_on_match_down'.");
463  }
464  else if ( wgm==8 || wgm==10 ) {
465  if ( p->config.compare0.output != HW_A1(hw_ocu_output_disconnected)
466  && p->config.compare0.output != HW_A1(hw_ocu_output_toggle_on_match)
467  && p->config.compare0.output != HW_A1(hw_ocu_output_clear_on_match_up_set_on_match_down)
468  && p->config.compare0.output != HW_A1(hw_ocu_output_set_on_match_up_clear_on_match_down) )
469  HWA_ERR("compare output A of class _c16a counter mode must be "
470  "'disconnected', 'toggle_on_match', "
471  "'clear_on_match_up_set_on_match_down', "
472  "or 'set_on_match_up_clear_on_match_down'.");
473  }
474  // else if ( p->top == HW_A1(hw_c16a_top_compare0)
475  else if ( p->config.compare0.output ) {
476  HWA_ERR("can not use of `compare_a` as both top value and output of class _c16a counter.");
477  }
478  }
479 
480  /* Check compare output B
481  */
482  if ( p->config.compare1.output != 0xFF ) {
483  if ( wgm==0 || wgm==4 || wgm==12 ) {
