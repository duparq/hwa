90 HW_INLINE void _hwa_docfusia( hwa_t *hwa, hwa_usia_t *p, uint8_t mode, uint8_t clock )
91 {
92  if ( mode != HW_A1(hw_usia_mode_spi_master)
93  && mode != HW_A1(hw_usia_mode_spi_slave) ) {
94  HWA_ERR("sorry, desired mode is not supported yet.");
95  return ;
96  }
97 
98  if ( clock != HW_A1(hw_usia_clock_software) ) {
99  HWA_ERR("sorry, desired clock mode is not supported yet.");
100  return ;
101  }
102 
103  if ( mode == HW_A1(hw_usia_mode_spi_master) ) {
104  hwa_config(hw_pin_usck, direction, output);
105  hwa_config(hw_pin_do, direction, output);
106  hwa_config(hw_pin_di, direction, input);
107 
108  _hwa_write_creg( p, _usia, wm, 1 );
109  _hwa_write_creg( p, _usia, cs, 2 );
110  _hwa_write_creg( p, _usia, clk, 1 );
111  }
112  else {
113  _hwa_write_creg( p, _usia, wm, 1 );
114  _hwa_write_creg( p, _usia, cs, 2 );
115  _hwa_write_creg( p, _usia, clk, 0 );
116  }
117 }
