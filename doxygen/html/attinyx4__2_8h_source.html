65 HW_INLINE void _hwa_commit_all( hwa_t *hwa )
66 {
67  /* Solve the configuration.
68  *
69  * The configuration values are written here since the solve function only
70  * has the address of the counter and does not know its name.
71  *
72  * _hw_creg(...) could be used as well though it can not access registers of
73  * another object.
74  */
75  _hwa_solve_c8a( &hwa->hw_counter0 );
76  if ( hwa->hw_counter0.solved.cs != 0xFF )
77  _hwa_write_reg( hw_counter0, cs, hwa->hw_counter0.solved.cs );
78  if ( hwa->hw_counter0.solved.wgm != 0xFF )
79  _hwa_write_reg( hw_counter0, wgm, hwa->hw_counter0.solved.wgm );
80  if ( hwa->hw_counter0.solved.coma != 0xFF )
81  _hwa_write_reg( hw_counter0, compare0_mode, hwa->hw_counter0.solved.coma );
82  if ( hwa->hw_counter0.solved.comb != 0xFF )
83  _hwa_write_reg( hw_counter0, compare1_mode, hwa->hw_counter0.solved.comb );
84 
85  _hwa_solve( hw_counter1 );
86  if ( hwa->hw_counter1.solved.cs != 0xFF )
87  _hwa_write_reg( hw_counter1, cs, hwa->hw_counter1.solved.cs );
88  if ( hwa->hw_counter1.solved.wgm != 0xFF )
89  _hwa_write_reg( hw_counter1, wgm, hwa->hw_counter1.solved.wgm );
90  if ( hwa->hw_counter1.solved.coma != 0xFF )
91  _hwa_write_reg( hw_counter1, compare0_mode, hwa->hw_counter1.solved.coma );
92  if ( hwa->hw_counter1.solved.comb != 0xFF )
93  _hwa_write_reg( hw_counter1, compare1_mode, hwa->hw_counter1.solved.comb );
94  if ( hwa->hw_counter1.solved.acic != 0xFF )
95  _hwa_write_reg( hw_counter1, acic, hwa->hw_counter1.solved.acic );
96  if ( hwa->hw_counter1.solved.ices != 0xFF )
97  _hwa_write_reg( hw_counter1, ices, hwa->hw_counter1.solved.ices );
98  if ( hwa->hw_counter1.solved.icnc != 0xFF )
99  _hwa_write_reg( hw_counter1, icnc, hwa->hw_counter1.solved.icnc );
100 
101 
102  /* If used, compare outputs must be configured as i/o outputs
103  */
104  if ( hwa->hw_counter0.config.compare0.output != 0xFF
105  && hwa->hw_counter0.config.compare0.output != HW_A1(hw_ocu_output_disconnected) )
106  hwa_config( hw_pin_pb2, direction, output );
107  if ( hwa->hw_counter0.config.compare1.output != 0xFF
108  && hwa->hw_counter0.config.compare1.output != HW_A1(hw_ocu_output_disconnected) )
109  hwa_config( hw_pin_pa7, direction, output );
110  if ( hwa->hw_counter1.config.compare0.output != 0xFF
111  && hwa->hw_counter1.config.compare0.output != HW_A1(hw_ocu_output_disconnected) )
112  hwa_config( hw_pin_pa6, direction, output );
113  if ( hwa->hw_counter1.config.compare1.output != 0xFF
114  && hwa->hw_counter1.config.compare1.output != HW_A1(hw_ocu_output_disconnected) )
115  hwa_config( hw_pin_pa5, direction, output );
116 
117  _hwa_commit( hw_core0 );
118  _hwa_commit( hw_wdog0 );
119  _hwa_commit( hw_porta );
120  _hwa_commit( hw_portb );
121 
122  _hwa_commit( hw_pcic0 );
123  /* _hwa_commit_r8( hwa, &hwa->hw_pcic0.msk ); */
124  /* _hwa_commit_r8( hwa, &hwa->hw_pcic1.msk ); */
125  /* _hwa_commit_reg_p( &hwa->hw_pcic0, _pcica0, msk ); */
126  /* _hwa_commit_reg_p( &hwa->hw_pcic1, _pcica1, msk ); */
127 
128  _hwa_commit( hw_counter0 );
129  _hwa_commit( hw_counter1 );
130  _hwa_commit( hw_usi0 );
131  _hwa_commit( hw_adc0 );
132  _hwa_commit( hw_acmp0 );
133 }
