54 HW_INLINE void _hw_eeproma_rdn( uint8_t *dst, intptr_t src, uint8_t n,
55  intptr_t ar,
56  intptr_t cr, uint8_t cb,
57  intptr_t dr)
58 {
59  while ( n-- ) {
60  *(volatile uint16_t*)ar = src++ ;
61  *(volatile uint8_t*)cr |= (1<<cb) ;
62  *dst++ = *(volatile uint8_t*) dr ;
63  }

84 HW_INLINE void _hw_wreeproma( uint16_t a, uint8_t v,
85  intptr_t ar,
86  intptr_t dr,
87  intptr_t mpe_ra, uint8_t mpe_bp,
88  intptr_t pe_ra, uint8_t pe_bp )
89 {
90  /* Assume EECR is 0, programming mode is 'Erase & write'
91  */
92  *(volatile uint16_t*)ar = a ;
93  *(volatile uint8_t*)dr = v ;
94  *(volatile uint8_t*)mpe_ra |= (1U<<mpe_bp) ;
95  *(volatile uint8_t*)pe_ra |= (1U<<pe_bp) ;
96 }

118 HW_INLINE void _hw_eeproma_wrn ( intptr_t dst,
119  uint8_t *src, uint8_t n,
120  intptr_t ar,
121  intptr_t dr,
122  intptr_t mpe_ra, uint8_t mpe_bp,
123  intptr_t pe_ra, uint8_t pe_bp )
124 {
125  /* Assume EECR is 0, programming mode is 'Erase & write' */
126 
127  while ( n-- ) {
128  /*
129  * Wait for completion of previous write
130  */
131  while ( *(volatile uint8_t*)pe_ra & (1U<<pe_bp) ) {}
132  /*
133  * Write byte
134  */
135  *(volatile uint8_t*)dr = *src++ ;
136  *(volatile uint16_t*)ar = dst++ ;
137  *(volatile uint8_t*)mpe_ra |= (1U<<mpe_bp) ;
138  *(volatile uint8_t*)pe_ra |= (1U<<pe_bp) ;
139  }
