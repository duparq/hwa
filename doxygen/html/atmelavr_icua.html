HWA
HardWare Abstractor
_icua

A class _icua object is a capture unit.

Name of the related counter
#define CAPTURE hw_counter0_capture0
#define COUNTER hw_sup( CAPTURE )
#if hw_bn(COUNTER) != 16
# error Only 16-bit counter is supported!
#endif
Name of the related i/o pin
#define COUNTER hw_counter0
#define CAPTURE hw_sub( COUNTER, capture0 )
#define INPUT hw_io( CAPTURE )
#if hw_id(INPUT) == hw_id(hw_pin_6)
# error pin 6 is already used
#endif
Configure the unit
hw_config( CAPTURE,
[input, pin_icp
| hw_acmp0, ]
[edge, falling
| rising, ]
[filter, on
| off ]
);

Note: currently, only edge configuration is implemented for the synchronous method.

hwa_config( CAPTURE,
input, pin_icp
| hw_acmp0,
edge, falling
| rising,
[filter, on
| off ]
);
