63 HW_INLINE void _hw_write_r8 ( intptr_t ra, uint8_t rwm, uint8_t rfm,
64  uint8_t bn, uint8_t bp, uint8_t v )
65 {
66 #if defined HWA_CHECK_ACCESS
67  if ( ra == ~0 )
68  HWA_ERR("invalid access");
69 #endif
70 
71 #if !defined HWA_NO_CHECK_USEFUL
72  if ( bn == 0 )
73  HWA_ERR("no bit to be changed?");
74 #endif
75 
76  /* Mask of bits to modify
77  */
78  uint8_t wm = (1U<<bn)-1 ;
79 
80 #if !defined HWA_NO_CHECK_LIMITS
81  if (v > wm)
82  HWA_ERR("value too high for bits number");
83 #endif
84 
85  wm <<= bp ;
86  v <<= bp ;
87 
88  /* Check that we do not try to set non-writeable bits
89  */
90  if ( (v & wm & rwm) != (v & wm) )
91  HWA_ERR("bits not writeable.");
92 
93  volatile uint8_t *p = (volatile uint8_t *)ra ;
94 
95  if ( ra < 0x40 &&
96  (wm==0x01 || wm==0x02 || wm==0x04 || wm==0x08 ||
97  wm==0x10 || wm==0x20 || wm==0x40 || wm==0x80) ) {
98  /*
99  * Just 1 bit to be written at C address < 0x40 (ASM address < 0x20): use
100  * sbi/cbi
101  */
102  if ( v )
103  *p |= wm ; /* sbi */
104  else {
105  if ( wm & rfm )
106  HWA_ERR("flag bit can only be cleared by writing 1 into it.");
107  *p &= ~wm ; /* cbi */
108  }
109  }
110  else {
111  /*
112  * Mask of bits to be read
113  * = bits that are writeable and not to be modified and not flags
114  */
115  uint8_t rm = rwm & ~wm & ~rfm ;
116 
117  if ( rm == 0 )
118  /*
119  * Nothing to be read, just write the new value
120  */
121  *p = v ;
122  else {
123  /*
124  * Read-modify-write
125  */
126 #if 0
127  /*
128  * FIXME: hw_write_reg(hw_counter1, ices, 1) produces an extra
129  * 'andi reg, 0x9F' though bit 5 is not writeable and should not be reset.
130  */
131  *p = (*p & rm) | (v & wm) ;
132 #else
133  /* FIX
134  */
135  uint8_t sm = wm & v ; /* what has to be set */
136  uint8_t cm = wm & (~v) ; /* what has to be cleared */
137  *p = (*p & ~cm) | sm ;
138 #endif
139  }
140  }
141 }

236 HW_INLINE void _hwa_commit_r8 ( hwa_r8_t *r, uint8_t rwm, uint8_t rfm, _Bool commit )
237 {
238  if ( !commit ) {
239  r->ovalue = (r->ovalue & r->omask & ~r->mmask) | (r->mvalue & r->mmask) ;
240  r->omask |= r->mmask ;
241  r->mmask = 0 ;
242  return ;
243  }
244 
245  volatile uint8_t *p = (volatile uint8_t *)r->a ;
246 
247  /* Mask of bits to be written:
248  * = bits that are writeable (rwm)
249  * AND that has been written (mmask)
250  * AND that really need to be modified (ovalue != mvalue)
251  */
252  uint8_t wm = rwm & r->mmask & ((r->ovalue ^ r->mvalue) | ~r->omask);
253 
254  if ( wm ) {
255  if ( (uintptr_t)p < 0x40 &&
256  (wm==0x01 || wm==0x02 || wm==0x04 || wm==0x08 ||
257  wm==0x10 || wm==0x20 || wm==0x40 || wm==0x80 ) ) {
258  /*
259  * Just 1 bit to be modified at C address < 0x40 (ASM address < 0x20): use
260  * instruction CBI or SBI
261  */
262  if ( wm & r->mvalue )
263  *p |= wm ; /* sbi */
264  else
265  *p &= ~wm ; /* cbi */
266  r->ovalue = (r->ovalue & ~wm) | (r->mvalue & wm) ;
267  r->omask |= wm ;
268  r->mmask = 0 ;
269  return ;
270  }
271 
272  /* Mask of bits to be read
273  * = bits that are not to be modified (mmask)
274  * AND not flags (rfm)
275  * AND that are not known (omask)
276  * AND that are writeable (rwm)
277  */
278  uint8_t rm = ~r->mmask & ~rfm & ~r->omask & rwm ;
279 
280  /* Read only if needed
281  */
282  if ( rm )
283  r->ovalue = *p ;
284 
285  /* Compute new value
286  */
287  r->ovalue = ((r->ovalue & ~wm) | (r->mvalue & wm)) & ~rfm ;
288 
289  /* Write new value
290  */
291  *p = r->ovalue | (rfm & r->mmask & r->mvalue) ;
292  }
293 
294  r->omask |= r->mmask ;
295  r->mmask = 0 ;
296 }

299 HW_INLINE void _hwa_commit_r16 ( hwa_r16_t *r, uint16_t rwm, uint16_t rfm, _Bool commit )
300 {
301  if ( !commit ) {
302  r->ovalue = (r->ovalue & r->omask & ~r->mmask) | (r->mvalue & r->mmask) ;
303  r->omask |= r->mmask ;
304  r->mmask = 0 ;
305  return ;
306  }
307 
308  volatile uint16_t *p = (volatile uint16_t *)r->a ;
309 
310  uint16_t wm = rwm & r->mmask & ((r->ovalue ^ r->mvalue) | ~r->omask);
311 
312  if ( wm ) {
313  /*
314  * Do not check for sbi/cbi for 16-bit access as since 8-bit AVRs do not
315  * have 16-bit configuration registers, it is unlikely that it would be
316  * useful.
317  */
318  uint16_t rm = ~r->mmask & ~rfm & ~r->omask & rwm ;
319 
320  if ( rm )
321  r->ovalue = *p ;
322 
323  r->ovalue = ((r->ovalue & ~wm) | (r->mvalue & wm)) & ~rfm ;
324 
325  *p = r->ovalue | (rfm & r->mmask & r->mvalue) ;
326  }
327 
328  r->omask |= r->mmask ;
329  r->mmask = 0 ;
330 }

364 HW_INLINE uint16_t _hw_atomic_read_r16 ( intptr_t ra, uint8_t rbn, uint8_t rbp )
365 {
366  volatile uint8_t *pl = (volatile uint8_t *)ra+0 ;
367  volatile uint8_t *ph = (volatile uint8_t *)ra+1 ;
368  uint16_t v ;
369  uint16_t m = (1UL<<rbn)-1 ;
370 
371  if ( (m & 0xFF) && (m >> 8) ) {
372  uint8_t s = _hw_read_reg(hw_core0,sreg);
373  hw_disable_interrupts();
374  uint8_t lb = *pl ;
375  _hw_write_reg(hw_core0,sreg,s);
376  uint8_t hb = *ph ;
377  v = (hb << 8) | lb ;
378  }
379  else if ( m & 0xFF )
380  v = *pl ;
381  else
382  v = (*ph)<<8 ;
383  return (v>>rbp) & m ;
384 }
