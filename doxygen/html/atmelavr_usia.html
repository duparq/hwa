HWA
HardWare Abstractor
_usia

A class _usia object is an USI.

Configure the USI
#define USI hw_usi0
hw_config( USI,
mode, disconnected
| spi_master
| spi_slave
| 2wire_master
| 2wire_slave,
clock, software
| hw_counter0_compare
| external_rising
| external_falling
);
Read one byte from USI
#define USI hw_usi0
uint8_t byte = hw_read( USI );
Send a byte over the USI
#define USI hw_usi0
uint8_t byte = '\n';
hw_write( USI, byte );
Status
uint8_t byte ;
hw_stat_t( USI ) st ; // Structure of USI status
st = hw_stat( USI ); // Read the status
if ( st.txc ) // Transmission complete?
byte = hw_read( USI );
Clocking
#define USI hw_usi0
uint8_t byte = hw_trigger( USI );
