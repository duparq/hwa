97 HW_INLINE void _hw_swuart0_config ( hwa_t *hwa __attribute__((unused)) )
98 {
99  /* Counter
100  */
101 #if defined HW_SWUART0_COUNTER
102  hwa_config( HW_SWUART0_COUNTER,
103  clock, HW_G2(syshz_div, HW_SWUART0_COUNTER_CLK_DIV),
104  countmode, loop_up,
105  top, max );
106 #endif
107 
108  /* RX pin
109  */
110 #if defined HW_SWUART0_PIN_RX
111  hwa_config( HW_SWUART0_PIN_RX, direction, input );
112  hwa_clear_irq( HW_SWUART0_PIN_RX, change );
113  hwa_turn_irq( HW_SWUART0_PIN_RX, change, on );
114  hwa_write_reg( hw_pcic0, HW_SWUART0_PIN_RX, 1 );
115 #endif
116 
117  /* TX pin
118  *
119  * Configure TX pin as state 1 output unless the pin is also used for RX.
120  */
121 #if defined HW_SWUART0_PIN_TX && hw_id(HW_SWUART0_PIN_TX) != hw_id(HW_SWUART0_PIN_RX)
122  hwa_config( HW_SWUART0_PIN_TX, direction, output );
123  hwa_write( HW_SWUART0_PIN_TX, 1 );
124 #endif
125 
126  /* DBG pin
127  */
128 #if defined HW_SWUART0_PIN_DBG
129  hwa_config( HW_SWUART0_PIN_DBG, direction, output );
130 #endif
131 }
