180 HW_INLINE void __hw_flasha_ldpgbf( void *src )
181 {
182  uint8_t r1 ;
183 
184 #if HW_DEVICE_FLASH_PAGE_SIZE > 256
185 # error
186 #endif
187 
188  hw_asm(
189  /* Clear page buffer
190  */
191  " ldi %[r1], 1<<BP_RWWSRE|1<<BP_SPMEN \n"
192  " .if CSR < 0x40 \n"
193  " out CSR, %[r1] \n"
194  " .else \n"
195  " sts CSR+0x20, %[r1] \n"
196  " .endif \n"
197 
198  /* Load page buffer
199  */
200  " clr r30 \n"
201  " clr r31 \n"
202  "1: ld r0, %a2+ \n"
203  " ld r1, %a2+ \n"
204 
205  " .if CSR < 0x20 \n"
206  " sbi CSR, BP_SPMEN \n"
207  " .else \n"
208  " ldi %[r1], 1<<BP_SPMEN \n"
209  " .if CSR < 0x40 \n"
210  " out CSR, %[r1] \n"
211  " .else \n"
212  " sts CSR+0x20, %[r1] \n"
213  " .endif \n"
214  " .endif \n"
215  " spm \n"
216 
217  " adiw r30, 2 \n"
218  " cpi r30, PGSIZE \n"
219  " brne 1b \n"
220  :
221  [r1] "=r" (r1),
222  "=e" (src)
223  :
224  "1" (src)
225  :
226  "r0", "r1",
227  "r30", "r31"
228  );
229 }
